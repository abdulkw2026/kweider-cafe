const allergens = [ 'Milk', 'Dairy', 'Cheese', 'Cream', 'Ghee', 'Yogurt', 'Butter', 'Sesame', 'Tahini', 'Nuts', 'Pistachio', 'Walnut', 'Almond', 'Pine Nuts', 'Mixed Nuts', 'Gluten', 'Wheat', 'Semolina', 'Bulgur', 'Soya', 'Soy', 'Eggs', 'Egg' ]; document.addEventListener("DOMContentLoaded", function() { const firstBtn = document.querySelector('.cat-btn'); /* السر هنا: أضفنا كلمة false لكي لا يقوم بالنزول التلقائي عند فتح الصفحة! سيقوم فقط بإظهار القسم الأول، وتبقى الصفحة في الأعلى. */ if (firstBtn) filterMenu('breakfast', firstBtn, false); const navBar = document.querySelector('.categories-bar'); const header = document.querySelector('header'); window.addEventListener('scroll', function() { if (window.scrollY > header.offsetHeight - 10) { navBar.classList.add('is-sticky'); } else { navBar.classList.remove('is-sticky'); } }); }); const modal = document.getElementById('detailModal'); const modalImgElement = document.getElementById('m-img'); function openDetails(el) { document.getElementById('m-ar').innerText = el.getAttribute('data-name-ar'); document.getElementById('m-en').innerText = el.getAttribute('data-name-en'); document.getElementById('m-desc-ar').innerText = el.getAttribute('data-desc-ar'); document.getElementById('m-desc-en').innerText = el.getAttribute('data-desc-en'); document.getElementById('m-price').innerHTML = el.getAttribute('data-price'); modalImgElement.src = el.getAttribute('data-img'); let allergyTxt = el.getAttribute('data-allergy'); let algHtml = 'Allergy Info:
' + allergyTxt; allergens.forEach(alg => { let r = new RegExp('(' + alg + ')', 'gi'); algHtml = algHtml.replace(r, '$1'); }); document.getElementById('m-allergy').innerHTML = algHtml; modal.style.display = 'flex'; } modalImgElement.addEventListener('click', function(e) { e.stopPropagation(); this.classList.toggle('img-zoomed'); }); modal.addEventListener('click', function() { modal.style.display = 'none'; modalImgElement.classList.remove('img-zoomed'); }); /* أضفنا shouldScroll للتحكم متى ينزل المتصفح ومتى يبقى ثابتاً */ function filterMenu(category, btn, shouldScroll = true) { const sections = document.querySelectorAll('.menu-section'); const buttons = document.querySelectorAll('.cat-btn'); buttons.forEach(b => b.classList.remove('active')); if (btn) { btn.classList.add('active'); } else { buttons.forEach(b => { if(b.getAttribute('onclick').includes(category)) { b.classList.add('active'); } }); } sections.forEach(section => { if (section.id === category) { section.style.display = 'block'; } else { section.style.display = 'none'; } }); // سينزل المتصفح لقسم المنتجات فقط إذا ضغط الزبون بيده (وليس عند التحميل الأول) if (shouldScroll) { const navBarHeight = document.querySelector('.categories-bar').offsetHeight; const offsetPosition = document.getElementById('main-menu').offsetTop - navBarHeight - 20; window.scrollTo({top: offsetPosition, behavior: 'smooth'}); } } function goToSection(categoryId) { filterMenu(categoryId, null, true); } let topButton = document.getElementById("backToTop"); window.onscroll = function() { if (document.body.scrollTop > 300 || document.documentElement.scrollTop > 300) { topButton.style.display = "block"; } else { topButton.style.display = "none"; } }; function scrollToTop() { window.scrollTo({top: 0, behavior: 'smooth'}); }
